cmake_minimum_required(VERSION 3.20)


project(Solution 
	LANGUAGES CXX	
	VERSION 1.0.0)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_EXTENSIONS OFF)
set(CMAKE_CXX_STANDARD_REQUIRED ON)


find_package(spdlog REQUIRED)
find_package(OpenCV REQUIRED)
find_package(VTK REQUIRED)


add_executable(xRayTracer
	"main.cpp"
 "geometry.h" "camera.h" "ray.h" "sampler.h" "integrator.h" "image.h" "scene.h" "light.h" "primitive.h")

 target_compile_definitions(xRayTracer
    PRIVATE 
	DATA_DIR="${PROJECT_SOURCE_DIR}/testdata/"
	TINYOBJLOADER_IMPLEMENTATION
	#CULLING
	kEpsilon=FLT_EPSILON
	kInfinity=FLT_MAX
)

 target_include_directories(xRayTracer
PRIVATE
	${OpenCV_INCLUDE_LIBS}
	)
 
target_link_libraries(xRayTracer 
PRIVATE 
	spdlog::spdlog
	${OpenCV_LIBS}
	${VTK_LIBRARIES}
	)
